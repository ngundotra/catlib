theorem catholic_rule_of_faith :
    -- (1) There exists revealed teaching not in Scripture
    (∃ (t : ApostolicTeaching), isRevealed t ∧ ¬inScripture t) ∧
    -- (2) For any canonical text, its canonicity was determined by a
